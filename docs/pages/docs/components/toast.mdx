import { Button } from 'nifty-ui';
import { ToastAction } from 'nifty-ui';
import { useToast } from 'nifty-ui';

<div>
  <Button
    variant="outline"
    onClick={() => {
      toast({
        title: 'Scheduled: Catch up ',
        description: 'Friday, February 10, 2023 at 5:57 PM',
        action: (
          <ToastAction variant="default" altText="Goto schedule to undo">
            Undo
          </ToastAction>
        ),
      });
    }}
  >
    Add to calendar
  </Button>
</div>
