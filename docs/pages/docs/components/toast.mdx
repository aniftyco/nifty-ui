# Toast

Give your users a Toast! PROSTE!

---

import { Button, ToastAction, useToast, Toaster } from 'nifty-ui';

export const ToastDemo = () => {
  const { toast } = useToast();

return (

<Button
  variant="outline"
  onClick={() => {
    toast({
      title: 'Scheduled: Catch up ',
      description: 'Friday, February 10, 2023 at 5:57 PM',
      action: (
        <ToastAction variant="default" altText="Goto schedule to undo">
          Undo
        </ToastAction>
      ),
    });
  }}
>
  Add to calendar
</Button>
); };

<div>
  <ToastDemo />
  <Toaster />
</div>
